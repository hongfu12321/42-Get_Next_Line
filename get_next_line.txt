#include "get_next_line.h"
#include "libft.h"
#include <stdio.h>

char	*my_strsearch(char *str, char c)
{
	int 	i;
	char	*tmp;

	i = 0;
	while (str[i])
		if (str[i++] == c)
			break;
	if (i == (int)ft_strlen(str) && str[i - 1] != '\n')
		return (0);
	tmp = ft_strnew(i - 1);
	tmp = ft_strncpy(tmp, str, i - 1);
	return (tmp);
}

int		get_next_line(int fd, char **line)
{
	static char	*buf;
	int			ret;
	char		*tmp;

	printf("----IN----\n");
	if ((fd < 0 || line == NULL))
		return (-1);
	if (!buf)
		buf = ft_strnew(BUFF_SIZE);

	*line = ft_strdup(buf);
	printf("\033[30;42mline_in: \033[0m%s\n", *line);
	ret = 100;
	while (!ft_strchr(*line, '\n')/* && (ret = read(fd, buf, BUFF_SIZE))*/)
	{
		ret = read(fd, buf, BUFF_SIZE);
		buf[ret] = '\0';
//		printf("buf_in_loop: %s\n", buf);
		printf("line_in: %s\n", *line);
		*line = ft_strjoin(*line, buf);
		if (!ft_strchr(*line, '\n'))
			printf("no newline!\n");
//		printf("line_after_join: %s\n", *line);
		if (ret != BUFF_SIZE)
			break;
	}
	if (ret == 0)
	{
		return (0);
	}
	buf = ft_strdup((ft_strchr(*line, '\n') + 1));
	tmp = *line;
	*line = ft_strdup(my_strsearch(*line, '\n'));
//	printf("line_out: %s\n", *line);
//	printf("buf_out: %s\n", buf);
	free(tmp);
//	printf("----OUT----\n");
	return (1);
}
